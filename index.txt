function getChecked(name){
  const otroMap = {
    demo: 'demo-otro-txt',
    prob: 'prob-otro-txt',
    serv: 'serv-otro-txt',
    enfoque: 'enfoque-otro-txt',
  };
  return Array.from(document.querySelectorAll(`input[name="${name}"]:checked`))
    .map(c => {
      if(c.value === '__otro__'){
        const txt = document.getElementById(otroMap[name]);
        return txt && txt.value.trim() ? 'Otro: ' + txt.value.trim() : '';
      }
      return c.value;
    })
    .filter(Boolean).join(', ');
}

function guardarLocal(p){
  let list = JSON.parse(localStorage.getItem('psi_reportes')||'[]');
  const idx = list.findIndex(r=>r.grupo===p.grupo);
  const entry = {...p, _ts: new Date().toISOString()};
  if(idx>=0) list[idx]=entry; else list.push(entry);
  localStorage.setItem('psi_reportes', JSON.stringify(list));
}
function cargarLocal(){
  return JSON.parse(localStorage.getItem('psi_reportes')||'[]');
}

let allReportes = [];

function cargarDatos(){
  allReportes = cargarLocal();
  if(CFG.SCRIPT_URL && CFG.SCRIPT_URL!=='PEGAR_URL_AQUI'){
    fetch(CFG.SCRIPT_URL+'?action=getAll')
      .then(r=>r.json())
      .then(data=>{
        if(data?.reportes?.length){
          const fromServer = data.reportes;
          const sgs = new Set(fromServer.map(r=>r.grupo));
          const localOnly = allReportes.filter(r=>!sgs.has(r.grupo));
          allReportes = [...fromServer,...localOnly];
        }
        renderPanel();
      }).catch(()=>renderPanel());
  } else { renderPanel(); }
}

function renderPanel(){
  actualizarStats();
  llenarFiltros();
  renderGrupos(allReportes);
}

function actualizarStats(){
  const n = allReportes.length;
  const comp = allReportes.filter(r=>r.comunidad&&r.modelo&&r.integrantes&&r.sintesis).length;
  let ses=0, links=0;
  allReportes.forEach(r=>{
    try{ses+=JSON.parse(r.sesiones||'[]').length}catch{}
    if(r.fotosComunidad) links++;
    if(r.plano) links++;
    if(r.pptx) links++;
    if(r.docsAdicionales) links++;
    try{ JSON.parse(r.sesiones||'[]').forEach(s=>{if(s.drive)links++}); }catch{}
  });
  document.getElementById('s-total').textContent=n;
  document.getElementById('s-comp').textContent=comp;
  document.getElementById('s-ses').textContent=ses;
  document.getElementById('s-links').textContent=links;
  actualizarHeroStats();
}

function llenarFiltros(){
  const modelos=[...new Set(allReportes.map(r=>r.modelo).filter(Boolean))].sort();
  const grupos=[...new Set(allReportes.map(r=>r.grupo).filter(Boolean))].sort();
  const sm=document.getElementById('doc-filter-modelo');
  sm.innerHTML='<option value="">Todos los modelos</option>';
  modelos.forEach(m=>{const o=document.createElement('option');o.value=m;o.textContent=m;sm.appendChild(o)});
  const sg=document.getElementById('doc-filter-grupo');
  sg.innerHTML='<option value="">Todos los grupos</option>';
  grupos.forEach(g=>{const o=document.createElement('option');o.value=g;o.textContent=g;sg.appendChild(o)});
}

function filtrar(){
  const q=document.getElementById('doc-search').value.toLowerCase();
  const fm=document.getElementById('doc-filter-modelo').value;
  const fg=document.getElementById('doc-filter-grupo').value;
  const fil=allReportes.filter(r=>{
    const t=`${r.grupo} ${r.comunidad} ${r.modelo} ${r.integrantes}`.toLowerCase();
    return (!q||t.includes(q))&&(!fm||r.modelo===fm)&&(!fg||r.grupo===fg);
  });
  renderGrupos(fil);
}

    return `<div class="grupo-card" onclick="abrirDetalle(${idx})">
      <div class="gc-head">
        <div class="gc-av ${cl}">${av}</div>
        <div class="gc-info">
          <div class="gc-nombre">${r.grupo||'Grupo sin nombre'}</div>
          <div class="gc-sub">${r.integrantes? trunc(r.integrantes,75):'Sin integrantes registrados'}</div>
        </div>
        <div class="gc-badges">
          ${r.modelo?`<span class="badge modelo">${trunc(r.modelo,32)}</span>`:''}
          ${fecha?`<span class="badge fecha">${fecha}</span>`:''}
          ${sesC?`<span class="badge ses">${sesC} sesión${sesC>1?'es':''}</span>`:''}
        </div>
      </div>
      <div class="gc-foot">
        <span class="gc-loc"><i class="ti ti-map-pin"></i>${r.comunidad? r.comunidad+(r.ciudad?', '+r.ciudad:''):'<em>Comunidad no especificada</em>'}</span>
        ${r.maps?`<a href="${r.maps}" target="_blank" class="btn-maps" onclick="event.stopPropagation()"><i class="ti ti-map"></i> Maps</a>`:''}
        <div class="gc-actions">
          <button class="btn-ver" onclick="abrirDetalle(${idx});event.stopPropagation()"><i class="ti ti-eye"></i> Ver reporte</button>
        </div>
      </div>
    </div>`;
  }).join('');
}

function trunc(s,n){ return s&&s.length>n? s.slice(0,n)+'…':(s||''); }

function df(lbl,val){
  const e=!val||!val.toString().trim();
  return `<div class="det-field"><div class="det-lbl">${lbl}</div><div class="det-val${e?' empty':''}">${e?'No registrado':val.toString().replace(/\n/g,'<br>')}</div></div>`;
}
function dft(lbl,val){
  if(!val||!val.trim()) return df(lbl,'');
  const items=val.split(',').map(s=>s.trim()).filter(Boolean);
  return `<div class="det-field"><div class="det-lbl">${lbl}</div><div class="det-tags">${items.map(t=>`<span class="det-tag">${t}</span>`).join('')}</div></div>`;
}

function cerrarDetalle(){ document.getElementById('det-overlay').classList.remove('open'); document.body.style.overflow=''; }
function clicarFuera(e){ if(e.target.id==='det-overlay') cerrarDetalle(); }
